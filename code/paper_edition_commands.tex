% Define conditional table input command
\newcommand{\smartinputtable}[2]{%
  \ifdefstring{\tableversion}{inline}{%
    % Inline version: place table here
    \singlespacing
    \input{#1}
    \setstretch{1.5}
  }{%
  }%
}

\newcommand{\smartinputtableend}[2]{%
  \ifdefstring{\tableversion}{inline}{%
    \input{#2}
  }{%
    \input{#1}
  }%
}
